Shader Anatomy!

A unity shader is split up into two main sections: "shaderlab" (configuration) part , and "CG" (hlsl) part. Make a new shader and paste this in, to play around with.

If you are familiar with html/css/js work, you can think shaderlab as html, HLSL as JS.

Shader "LearnSurface/Color"
{
    Properties
    {
        _Color ("Color", Color) = (1,1,1,1)
        _MainTex ("Albedo (RGB)", 2D) = "white" {}
        _Glossiness ("Smoothness", Range(0, 1)) = 0.5
        _Metallic ("Metallic", Range(0, 1)) = 0.0
    }
    SubShader
    {
        Tags { "RenderType"="Opaque" }
		
        ZWrite On
        Cull Back
        Blend Off
		
        CGPROGRAM	
        #pragma surface surf Standard fullforwardshadows vertex:vert addshadow dithercrossfade
        #pragma target 5.0
        
        struct Input
        {
            float2 texcoord;
        };
		
        float4 _Color;			
        sampler2D _MainTex; 	
        float _Glossiness;  		
        float _Metallic;    	
        
        void vert(inout appdata_full app, out Input input)
        {
            UNITY_INITIALIZE_OUTPUT(Input, input);
            input.texcoord = app.texcoord.xy;
        }

        void surf(Input input, inout SurfaceOutputStandard o)
        {
            o.Albedo = tex2D(_MainTex, input.texcoord).rgb * _Color.rgb;
            o.Smoothness = _Glossiness;
            o.Metallic = _Metallic;
            o.Normal = float3(0, 0, 1);
        }	
        ENDCG
    }
    FallBack "Diffuse"
}


				
				
				

struct appdata_full
{
    float4 vertex : POSITION;
    float4 tangent : TANGENT;
    float3 normal : NORMAL;
    float4 color : COLOR;
    float4 texcoord : TEXCOORD0;
    float4 texcoord1 : TEXCOORD1;
    float4 texcoord2 : TEXCOORD2;
    float4 texcoord3 : TEXCOORD3;
    float4 texcoord4 : TEXCOORD4;
    float4 texcoord5 : TEXCOORD5;
};

struct Input
{
    float3 viewDir;
    float4 screenPos;
    float3 worldPos;
    float3 worldRefl;
    float3 worldNormal;
};

struct SurfaceOutputStandard
{
    float3 Albedo;      // Base (diffuse or specular) color
    float3 Normal;      // Tangent space normal, if written
    float3 Emission;    // Emission color
    float Metallic;     // 0 = non-metal, 1 = metal
    float Smoothness;   // 0=  rough, 1 = smooth
    float Occlusion;    // Occlusion (default 1)
    float Alpha;        // Alpha for transparencies
};
